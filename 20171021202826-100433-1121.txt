#include<iostream>
using namespace std;
int n,A[10]={1};
int main()
{
    for(int i=1;i<=9;i++) A[i]=i*A[i-1]; //½×³Ë±í
    while(cin>>n&&n>=0)
    {
        if(n==0){cout<<"NO"<<endl;continue;}
        for(int i=9;i>=0&&n;i--) if(A[i]<=n) n-=A[i];
        n==0?cout<<"YES"<<endl:cout<<"NO"<<endl;
    }
    return 0;
}
