#include<iostream>
#include<cstdio>
#include<cstring>
using namespace std;
const int MAXN = 10;
int data[MAXN];
int n,k;
int num=0;
void dfs(int ans)///ans是递归的层数
{
    if(ans==k)
    {
        printf("%d\n",num);
        return ;
    }
    for(int i=1; i<=n; i++)
    {
        num=num*10+i;
        dfs(ans+1);///内嵌套for
        num=(num-i)/10;
    }
}
int main()
{
    while(~scanf("%d%d",&n,&k))
    {
        num=0;
        for(int i=1; i<=n; i++)
        {
            data[i]=i;
        }
        dfs(0);
    }
return 0;
}
