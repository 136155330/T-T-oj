#include<stdio.h>
#include<math.h>
main()
{
    int t,i,a,b,c,s,j;
    while(~scanf("%d",&t))
    {s=0;
        for(i=0;i<t;i++)
        {
            scanf("%d(%d)",&a,&b);
            {j=0;
                while(a)
            {c=a%10;
            s+=c*pow(b,j++);
              a/=10;
            }
            }
        }
        printf("%d\n",s);
    }
}
