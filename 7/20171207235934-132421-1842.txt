#include<stdio.h>
int x[1005];
int main()
{
    int n;int i;int j;int t;
    while(scanf("%d",&n)!=EOF)///循环输入
    {
        for(i=0;i<n;i++) ///0.....n-1 || n
            scanf("%d",&x[i]);
        for(j=n-1;j!=-1;j--)
        {
            for(i=0;i<j/*n-1*/;i++)///n-2 n-1+1=n
            {
              if(x[i]>=x[i+1])///n-1  3000个数字排序
              {
                  t=x[i+1];
                  x[i+1]=x[i];
                  x[i]=t;
              }

            }
        }for(i=0;i<n;i++)
            printf("%d ",x[i]);
        printf("\n");

    }
    return 0;
}
