#include<stdio.h>
/*
(a + b) % p = (a % p + b % p) % p £¨1£©
(a - b) % p = (a % p - b % p) % p £¨2£©
(a * b) % p = (a % p * b % p) % p £¨3£©
a ^ b % p = ((a % p)^b) % p £¨4£©
*/
typedef long long ll;
ll quickpow(ll a,ll b,ll k)
{
    ll ans=0;
    ll temp=a%k;
    while(b!=0)
    {
        if(b&1!=0)
        {
            ans=(ans+temp%k)%k;
        }
        temp=(temp%k+temp%k)%k;
        b>>=1;
    }
    return ans;
}
int main()
{
ll n,c,ans,tmp;
while(~scanf("%lld%lld",&n,&c))
{
        ans=1;
        while(n--)
        {
            scanf("%lld",&tmp);
            ans=quickpow(ans,tmp,c);
        }
        printf("%lld\n",ans);
}
return 0;
}
