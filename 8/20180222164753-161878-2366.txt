int x[100005];
int findup(int x[],int math,int n)
{
    int left,right,mid;
    left=-1;
    right=n;
    mid=(right+left)/2;
    while(left+1<right)
    {
        if(x[mid]>math)
        {
            right=mid;
            mid=(right+left)/2;
        }
        else if(x[mid]<=math)
        {
            left=mid;
            mid=(right+left)/2;
        }
    }
    return right;///返回的是比它大的值的下标
}
int finddown(int x[],int math,int n)
{
    int left,right,mid;
    left=-1;
    right=n;
    mid=(right+left)/2;
    while(left+1<right)
    {
        if(x[mid]>=math)
        {
            right=mid;
            mid=(right+left)/2;
        }
        else if(x[mid]<math)
        {
            left=mid;
            mid=(right+left)/2;
        }
    }
    ///返回的是比他小的值的下标
    return left;
}
