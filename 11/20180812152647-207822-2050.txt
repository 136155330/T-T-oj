#include<iostream>

using namespace std;
const int maxn = 1e5 + 5;
const int sl = 30;
int n,k;
struct Trie
{
    int next[sl];
    int cnt;/// Ù–‘”Ú
}trie[maxn*sl];
int sizes;
void Init()
{
    memset(trie[0].next,0,sizeof(trie[0].next));
    trie[0].cnt = 0;
    sizes = 1;
}
void insert(char * s, int num)
{
    int now = 0;
    int len = (int)strlen(s);
    for (int i = 0; i < len; i ++) {
        int ans = *(s+i) - 'a';
        //printf("ans = %d\n",ans);
        if(trie[now].next[ans] == 0)
        {
            memset(trie[sizes].next, 0, sizeof(trie[sizes].next));
            trie[sizes].cnt = 0;
            trie[now].next[ans] = sizes ++;
            //printf("%d %d\n",trie[now].next[ans],now);
        }
        now = trie[now].next[ans];
    }
    trie[now].cnt = num;
    //printf("now %d\n",num);
}
int finds(char * s)
{
    int now = 0;
    int len = (int)strlen(s);
    int i;
    for (i = 0; i < len; i ++) {
        int ans = *(s + i) - 'a';
        //printf("num = %d\n",trie[now].next[ans]);
        if (trie[now].next[ans]) {
            now = trie[now].next[ans];
            //printf("now %d\n",now);
        }
        else
        {
            //printf("%d %d\n",now,ans);
            return -1;
        }
    }
    if(i == len)
    {
        //printf("now %d\n",now);
        return trie[now].cnt;
    }
    return -1;
}
