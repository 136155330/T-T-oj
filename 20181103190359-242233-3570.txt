const int MAXN=1e3+5;

void dfs()
{
    int q,w;
    int num=0;
    for(int i=0; i<z; i++)
        str2[i]=c[1][i];
    str2[z]='\0';
    for(int i=1; i<=n; i++)
    {
        for(int j=0; j<m; j+=z)
        {
            for(int k=0; k<z; k++)
                str1[k]=c[i][j+k];
            str1[z]='\0';
            if(strcmp(str1,str2)!=0)
            {
                num++;
                q=i;
                w=j;
            }
        }
    }
    if(num==1)
        printf("%d %d\n",q,w/3+1);
    else
        printf("%d %d\n",1,1);
}
