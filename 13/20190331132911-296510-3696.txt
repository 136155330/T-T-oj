#include <bitset>
#include <iostream>
#include <cstdio>
#include <cstring>
using namespace std;
/**
 万万没想到这是个DP。
 我们可以使用一个bitset来保持某个位置是否有1
 然后把这个值移动j * j位
 那么就可以得到对应的值
 最多就是移动100的100
 100000
 **/
const int MAXN = 1000000 + 5;
bitset<MAXN>s[105];
void Init()
{
    for(int i = 0; i < 105; i ++)
    {
        s[i].reset();
    }
}
int n;
int main()
{
    while(~scanf("%d", &n))
    {
        Init();
        s[0] = 1;
        for(int i = 1; i <= n; i ++)
        {
            int l, r;
            scanf("%d%d", &l, &r);
            for(int j = l; j <= r; j ++)
            {
                s[i] |= (s[i - 1] << (j * j));
            }
        }
        printf("%d\n", s[n].count());
    }
    return 0;
}
