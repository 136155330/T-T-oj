#include<stdio.h>
#include<string.h>
long long mod;
long long quickpow(long long a,int n)///是个快速幂
{
    long long ans=1;
    while(n)
    {
        if(n&1)
        {
            ans=ans*a%mod;
        }
        a=a*a%mod;
        n>>=1;
    }
    return ans;
}
char b[1000005];
long long quickpow10(long long a,int n)
{
    long long ans=1;
    for(int i=n-1;i>=0;i--)
    {
        ans=ans*quickpow(a,b[i]-'0')%mod;
        a=quickpow(a,10);
    }
    return ans;
}

int main()
{
    int i,j,a,n;
    while(~scanf("%d%s%I64d",&a,&b,&mod))
    {
    printf("%I64d\n",quickpow10(a,strlen(b)));
    }
    return 0;
}
