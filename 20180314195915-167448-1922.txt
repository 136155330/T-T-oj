#include<stdio.h>
#include<string.h>
char str[2000020];
char stk[2000020];
//ÑùÀý£º<<<<<<
int main()
{
    while(scanf("%s",str)!=EOF)
    {
        int top=0;
        int len=strlen(str);
        int flag=0;
        for(int i=0;i<len;i++)
        {
            if(str[i]=='('||str[i]=='{'||str[i]=='['||str[i]=='<')
            {
                stk[top++]=str[i];
            }
            else
            {
                if(top==0)
                {
                    puts("NO");
                    flag=1;
                    break;
                }
                if(str[i]==')')
                {
                    if(stk[top-1]=='(')
                        top--;
                    else
                    {
                        puts("NO");
                        flag=1;
                        break;
                    }
                }
                else if(str[i]=='}')
                {
                    if(stk[top-1]=='{')
                        top--;
                    else
                    {
                        puts("NO");
                        flag=1;
                        break;
                    }
                }
                else if(str[i]==']')
                {
                    if(stk[top-1]=='[')
                        top--;
                    else
                    {
                        puts("NO");
                        flag=1;
                        break;
                    }
                }
                else if(str[i]=='>')
                {
                    if(stk[top-1]=='<')
                        top--;
                    else
                    {
                        puts("NO");
                        flag=1;
                        break;
                    }
                }
            }
        }
        if(flag==0)
        {
            if(top==0)
                puts("YES");
            else puts("NO");
        }
    }
    return 0;
}
