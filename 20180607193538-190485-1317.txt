#include <iostream>
#include <cstdio>
#include <cstring>
#include <cmath>
using namespace std;
///¹«Ê½£ºf(n)=ln(n)+C+1/(2*n);
const double r = 0.57721566490153286060651209;
double math[10000];

int main()
{
    math[1] = 1;
    for (int i = 2; i < 10000; i++)
    {
        math[i] = math[i - 1] + 1.0 / i;
    }
    int n;
    int cases = 0;
    int t;
    scanf("%d", &t);
    while (t--)
    {
        scanf("%d", &n);
        if (n < 10000)
        {
            printf("Case %d: %.10lf\n", ++cases, math[n]);
        }
        else
        {
            printf("Case %d: %.10lf\n", ++cases, log(n) + 1.0 / (2 * n) + r);
        }
    }
    return 0;
}
