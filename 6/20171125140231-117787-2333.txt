#include<stdio.h>
/*这题是巴什博奕（Bash Game）什么是巴什博弈
我是这样想的，如果说如果说这个值是（m+1）*k+a,a<m；
那么就能保证我们拿的时候先手拿a，下一个那（1，m）由于
留下来的只是（m+1-先手拿的）必定小于m所以我们可以不断的够造一个
留下来的（m+1）给对手，这样的话我们必赢！
接下来是概念解释
只有一堆n个物品，两个人轮流从这堆物品中取物，规定每次至少取一个，最多取m个。最后取光者得胜。
 显然，如果n=m+1，1那么由于一次最多只能取m个，
 所以，无论先取者拿走多少个，后取者都能够一次拿走剩余的物品，后者取胜。
 因此我们发现了如何取胜的法则：每个回合时m+1个，如果n=（m+1)*r+s，（r为任意自然数，s≤m),那么先取者要拿走s个物品，
 如果后取者拿走k（≤m)个，那么先取者再拿走m+1-k个，结果剩下（m+1）（r-1）个，以后保持这样的取法，那么先取者肯定获胜
 。总之，要保持给对手留下（m+1）的倍数，就能最后获胜。
*/
int main()
{
    int t;int n,m;
    while(~scanf("%d",&t))
    {
        while(t--)
        {
            scanf("%d%d",&n,&m);
            if(n%(m+1)<=m&&n%(m+1)!=0)
                printf("YES\n");
            else
                printf("NO\n");
        }
    }
    return 0;
}
