#include<stdio.h>
#include<stdlib.h>
#define maxn 500005
int sum[maxn];
int max(int a,int b)
{
    if(a>b)return a;
    else
        return b;
}
int min(int a,int b)
{
    if(a<b)return a;
    else
        return b;
}
void ss()
{
    for(int i=0;i<maxn;i++) sum[i] = 1;
    for(int i=2;i<maxn;i++){
        for(int j=i;j<maxn;j+=i){
            sum[j] += i;
        }
    }
    sum[0] =0;
    for(int i=1;i<maxn;i++){
        sum[i] += sum[i-1];
    }
}
int main()
{
int t;int a;int b;int c;
ss();
while(~scanf("%d",&t))
{
    while(t--)
    {
        scanf("%d %d",&a,&b);
        c=sum[max(a,b)]-sum[min(a,b)-1];
        //printf("%d\n",c);
        if(c%2==0)
            printf("0\n");
        if(c%2!=0)
            printf("1\n");
    }
}
return 0;
}
